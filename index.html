🎮 Угадай число

Я загадал число от 1 до 100. Попробуй угадать!
from pyscript import Element import random number = random.randint(1, 100) attempts = 0 def check_guess(*args): global attempts input_elem = Element("guess-input") output_elem = Element("output") try: guess = int(input_elem.value) attempts += 1 if guess < 1 or guess > 100: output_elem.write("⚠️ Число должно быть от 1 до 100!") elif guess < number: output_elem.write(f"🔽 Загаданное число БОЛЬШЕ! (попытка {attempts})") elif guess > number: output_elem.write(f"🔼 Загаданное число МЕНЬШЕ! (попытка {attempts})") else: output_elem.write(f"🎉 ПОБЕДА! Число {number}, попыток: {attempts}") # Сброс для новой игры (если хочешь) # number = random.randint(1, 100) # attempts = 0 except: output_elem.write("❌ Ошибка! Введи число.") input_elem.clear()
